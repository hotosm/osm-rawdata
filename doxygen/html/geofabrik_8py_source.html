38class GeoFabrik(object):
39 def __init__(self):

50 def dump(self):
51 for entry in self.regions:
52 [[k, v]] = entry.items()
53 print(f"Region: {k}")
54 if isinstance(v, list):
55 for i in v:
56 print(f"\t{i}")

59 def getRegion(self, region: str):
60 for entry in self.regions:
61 [[k, v]] = entry.items()
62 if isinstance(v, list):
63 for i in v:
64 if i.lower() == region.lower():
65 return k
66 return None

69def download_file(url, dest):
70 try:
71 with requests.get(url, stream=True) as r:
72 r.raise_for_status()
73 total_size = int(r.headers.get("content-length", 0))
74 with open(dest, "wb") as f:
75 downloaded = 0
76 for chunk in r.iter_content(chunk_size=8192):
77 if chunk:
78 f.write(chunk)
79 downloaded += len(chunk)
80 percent_done = (downloaded / total_size) * 100
81 print(f"\rDownloading: {percent_done:.2f}%", end="")
82 print("\nDownload completed.")
83 except Exception as e:
84 log.error(f"Failed to download {url}: {e}")
